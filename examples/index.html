Usage:

Expected microformat is: 


<span class="money">
  <abbr class="unit">&euro;</abbr>
  <span class="amount">123.45</span>
  <abbr class="currency" title="EUR"></abbr>
</span>

Configure once, then run it as many times as you want: 


// Configure rates and default settings
jQuery.currency.configure({
  rates: {
    // Object containing exchange rates from the base currency (EUR by default)
    "USD": 1.2491,
    "GBP": 0.8032,
    "JPY": 99.8494
  }
});

// Convert all currencies in element with id="container" to USD
jQuery("#container").currency("USD");
